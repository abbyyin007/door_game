            💡 <strong>EXACT MIT RULES (from the text):</strong><br>

            • Each click costs 1 of 100 total clicks.<br>

            • First click on any door: opens it ($0).<br>

            • <strong>Consecutive clicks on the same door</strong> → fluctuating cash (ONE door always pays higher than the others).<br>

            • <strong>Switching to a different door</strong> → costs a click, gives $0 (even if that door was opened before).<br>

            • ⏰ If you ignore a door for 6 seconds, it shrinks and VANISHES forever.<br>

            • 🧠 Optimal strategy: test each door once (3 clicks, earn $0), find the high‑value door, then click ONLY that door for the remaining 97 clicks. But the fear of losing options makes people switch and waste clicks – that's the "inner paralysis".

<script>

    // ----- GAME STATE -----

    let clicksRemaining = 100;

    let totalEarned = 0;

    let gameActive = true;

    let lastClickedDoor = -1;          // index of last clicked door, -1 = none yet


    let doorOpened = [false, false, false];

    let doorActive = [true, true, true];

    let shrinkTimeouts = [null, null, null];

    let intervalIds = [null, null, null];

    let doorElements = [];

    let doorsData = []; // each: { clickCount, lastClickTime }


    const VANISH_DELAY_MS = 6000;


    // ----- REWARD STRUCTURE (hidden from player) -----

    // Primary door (index 0) ALWAYS gives more than secondary doors (index 1 & 2)

    const PRIMARY_MIN = 15;

    const PRIMARY_MAX = 25;

    const SECONDARY_MIN = 1;

    const SECONDARY_MAX = 10;  // 10 < 15, so primary always strictly higher


    function getClickReward(doorIndex) {

        // Only called when it's a consecutive click on same opened door

        if (doorIndex === 0) {

            return Math.floor(Math.random() * (PRIMARY_MAX - PRIMARY_MIN + 1)) + PRIMARY_MIN;

        } else {

            return Math.floor(Math.random() * (SECONDARY_MAX - SECONDARY_MIN + 1)) + SECONDARY_MIN;

        }

    }


    // Floating message animation (cash or warning)

    function showFloatingMessage(x, y, text, isCash = true) {

        const div = document.createElement('div');

        div.className = 'floating';

        div.innerText = text;

        div.style.color = isCash ? 'gold' : '#ffaa88';

        div.style.left = `${x - 20}px`;

        div.style.top = `${y - 30}px`;

        document.body.appendChild(div);

        setTimeout(() => div.remove(), 800);

    }


    // Update door scale (shrinking effect based on time since last click)

    function updateDoorScale(doorIndex) {

        const el = doorElements[doorIndex];

        if (!el || !doorActive[doorIndex]) return;

        const lastClick = doorsData[doorIndex]?.lastClickTime || 0;

        const elapsed = Date.now() - lastClick;

        let remaining = Math.max(0, VANISH_DELAY_MS - elapsed);

        let scale = remaining / VANISH_DELAY_MS;

        scale = Math.max(0, Math.min(1, scale));

        el.style.transform = `scale(${scale})`;

    }


    // Update timer bar and warning text

    function updateTimerBar(doorIndex) {

        const el = doorElements[doorIndex];

        if (!el || !doorActive[doorIndex]) return;

        const bar = el.querySelector('.timer-bar');

        if (!bar) return;

        const lastClick = doorsData[doorIndex]?.lastClickTime || 0;

        const elapsed = Date.now() - lastClick;

        const percent = Math.max(0, Math.min(100, (elapsed / VANISH_DELAY_MS) * 100));

        bar.style.width = `${100 - percent}%`;


        let warningEl = el.querySelector('.shrink-warning');

        if (!warningEl) {

            warningEl = document.createElement('div');

            warningEl.className = 'shrink-warning';

            el.appendChild(warningEl);

        }

        if (!doorActive[doorIndex] || !gameActive) {

            warningEl.style.display = 'none';

            return;

        }

        if (elapsed > VANISH_DELAY_MS - 1500) {

            warningEl.style.display = 'block';

            const secondsLeft = ((VANISH_DELAY_MS - elapsed) / 1000).toFixed(1);

            warningEl.innerText = `⚠️ vanishes in ${secondsLeft}s ⚠️`;

        } else {

            warningEl.style.display = 'none';

        }

    }


    function updateDoorVisuals() {

        for (let i = 0; i < 3; i++) {

            const el = doorElements[i];

            if (!el) continue;

            if (!doorActive[i]) {

                el.classList.add('vanished');

                continue;

            } else {

                el.classList.remove('vanished');

            }

            const rewardSpan = el.querySelector('.reward-value');

            if (!doorOpened[i]) {

                rewardSpan.innerText = `❓ ???`;

            } else {

                rewardSpan.innerText = `💰 ? ?`;

            }

            updateTimerBar(i);

            updateDoorScale(i);

        }

        document.getElementById('clicksLeft').innerText = clicksRemaining;

        document.getElementById('totalCash').innerText = totalEarned;

    }


    function scheduleVanish(doorIndex) {

        if (shrinkTimeouts[doorIndex]) clearTimeout(shrinkTimeouts[doorIndex]);

        if (!doorActive[doorIndex]) return;

        if (!gameActive) return;


        shrinkTimeouts[doorIndex] = setTimeout(() => {

            if (!gameActive) return;

            if (!doorActive[doorIndex]) return;

            const last = doorsData[doorIndex]?.lastClickTime || 0;

            if (Date.now() - last < VANISH_DELAY_MS - 100) {

                scheduleVanish(doorIndex);

                return;

            }

            doorActive[doorIndex] = false;

            const doorDiv = doorElements[doorIndex];

            if (doorDiv) doorDiv.classList.add('vanished');

            if (intervalIds[doorIndex]) clearInterval(intervalIds[doorIndex]);

            shrinkTimeouts[doorIndex] = null;

            const infoDiv = document.querySelector('.info-text');

            const original = infoDiv.innerHTML;

            infoDiv.innerHTML = `💔 DOOR ${['🔴 A', '🟢 B', '🔵 C'][doorIndex]} VANISHED! You lost that option forever. 💔<br>` + original;

            setTimeout(() => {

                if (infoDiv.innerHTML.includes("VANISHED")) infoDiv.innerHTML = original;

            }, 2000);

            updateDoorVisuals();

        }, VANISH_DELAY_MS);

    }


    function refreshDoorTimer(doorIndex) {

        if (!doorActive[doorIndex]) return;

        doorsData[doorIndex].lastClickTime = Date.now();

        if (shrinkTimeouts[doorIndex]) clearTimeout(shrinkTimeouts[doorIndex]);

        scheduleVanish(doorIndex);

        updateDoorScale(doorIndex);

        updateTimerBar(doorIndex);

    }


    function startUpdater(doorIndex) {

        if (intervalIds[doorIndex]) clearInterval(intervalIds[doorIndex]);

        intervalIds[doorIndex] = setInterval(() => {

            if (!doorActive[doorIndex] || !gameActive) return;

            updateTimerBar(doorIndex);

            updateDoorScale(doorIndex);

        }, 80);

    }


    // Core click logic with switch penalty

    function onDoorClick(doorIndex, event) {

        if (!gameActive) return;

        if (clicksRemaining <= 0) {

            alert("⛔ No clicks left! Game over. Press RESET.");

            return;

        }

        if (!doorActive[doorIndex]) return;


        const rect = event.currentTarget.getBoundingClientRect();

        const clickX = rect.left + rect.width / 2;

        const clickY = rect.top + 20;


        // Determine if this is a switch (different door than last click)

        const isSwitch = (lastClickedDoor !== -1 && lastClickedDoor !== doorIndex);

        let reward = 0;


        // Spend the click

        clicksRemaining--;


        // First click on this door?

        const firstTimeOnThisDoor = !doorOpened[doorIndex];

        if (firstTimeOnThisDoor) {

            doorOpened[doorIndex] = true;

            reward = 0;

            showFloatingMessage(clickX, clickY, "OPENED (0)", false);

        } else {

            // Door already opened before

            if (isSwitch) {

                // Switching costs a click and yields no cash

                reward = 0;

                showFloatingMessage(clickX, clickY, "SWITCH → $0", false);

            } else {

                // Consecutive click on same door -> earn cash

                reward = getClickReward(doorIndex);

                totalEarned += reward;

                showFloatingMessage(clickX, clickY, `+${reward}`, true);

            }

        }


        // Update door stats

        doorsData[doorIndex].clickCount++;

        refreshDoorTimer(doorIndex);

        lastClickedDoor = doorIndex;


        // Update click counter on UI

        const clickSpan = doorElements[doorIndex].querySelector('.click-count');

        clickSpan.innerText = `🖱️ clicks: ${doorsData[doorIndex].clickCount}`;


        // Slight visual feedback for cash vs switch

        const rewardBadge = doorElements[doorIndex].querySelector('.reward-badge');

        if (reward > 0) {

            rewardBadge.style.transform = "scale(1.1)";

            setTimeout(() => { if (rewardBadge) rewardBadge.style.transform = ""; }, 200);

        } else if (isSwitch && !firstTimeOnThisDoor) {

            rewardBadge.style.transform = "scale(0.95)";

            setTimeout(() => { if (rewardBadge) rewardBadge.style.transform = ""; }, 150);

        }


        updateDoorVisuals();


        if (clicksRemaining === 0) {

            gameActive = false;

            alert(`🏁 Game finished! Total cash: $${totalEarned}\n\nOptimal strategy would earn much more. Every switch wasted a potential earning click. Did you keep switching? That's the inner paralysis.`);

        }


        if (!doorActive.some(v => v === true) && clicksRemaining > 0) {

            gameActive = false;

            alert("All doors vanished! You let them shrink away. Reset to try again.");

        }

    }


    function resetGame() {

        for (let i = 0; i < shrinkTimeouts.length; i++) {

            if (shrinkTimeouts[i]) clearTimeout(shrinkTimeouts[i]);

            if (intervalIds[i]) clearInterval(intervalIds[i]);

        }

        clicksRemaining = 100;

        totalEarned = 0;

        gameActive = true;

        lastClickedDoor = -1;

        doorOpened = [false, false, false];

        doorActive = [true, true, true];


        const now = Date.now();

        doorsData = [

            { clickCount: 0, lastClickTime: now },

            { clickCount: 0, lastClickTime: now },

            { clickCount: 0, lastClickTime: now }

        ];


        const container = document.getElementById('doorsArea');

        container.innerHTML = '';

        doorElements = [];

        const doorNames = ["🔴 DOOR A", "🟢 DOOR B", "🔵 DOOR C"];


        for (let i = 0; i < 3; i++) {

            const doorDiv = document.createElement('div');

            doorDiv.className = 'door-card';

            doorDiv.setAttribute('data-idx', i);

            doorDiv.innerHTML = `

                <div class="door-title">${doorNames[i]}</div>

                <div class="reward-badge reward-value">❓ ???</div>

                <div class="click-count">🖱️ clicks: 0</div>

                <div class="timer-bar-container"><div class="timer-bar"></div></div>

            `;

            doorDiv.addEventListener('click', (e) => {

                const idx = parseInt(doorDiv.getAttribute('data-idx'));

                onDoorClick(idx, e);

            });

            container.appendChild(doorDiv);

            doorElements.push(doorDiv);

            startUpdater(i);

            scheduleVanish(i);

        }

        updateDoorVisuals();


        if (window._interval) clearInterval(window._interval);

        window._interval = setInterval(() => {

            if (!gameActive && clicksRemaining === 0) return;

            updateDoorVisuals();

        }, 150);

    }


    document.getElementById('resetBtn').addEventListener('click', () => {

        if (window._interval) clearInterval(window._interval);

        resetGame();

    });


    resetGame();

</script>
